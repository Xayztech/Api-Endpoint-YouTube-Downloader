API Key Manager

How To Use Guide

Complete tutorial for using Xayz API.

1. Generate API Key

Go to the API Key Manager menu. Click the "GENERATE NEW KEY" button. Copy the key that appears (e.g., xayztech-xxxx). This key is required for all requests.

2. Searching Videos

Go to the Search Engine menu. Enter your API Key and a Keyword (e.g., "Justin Bieber"). Click "GO". You can copy the link from the results.

3. Downloading Media

Go to Home Downloader. Paste your API Key and the YouTube URL. Click "START PROCESS". Buttons for MP4 (Video) and MP3 (Audio) will appear. Click to download directly.

4. Integration (Developers)

If you are a developer, check the Code Methods and Examples menus to see how to integrate this API into your Node.js, Python, or PHP applications.

Node.js (CommonJS)

server-side

Standard for older Node.js projects. Uses `require` and `axios`.

const axios = require('axios');

const CONFIG = {
    baseUrl: 'https://xayz-ytdl.vercel.app/api/youtube',
    apiKey: 'xayztech-YOUR_KEY_HERE',
    videoUrl: 'https://www.youtube.com/watch?v=dQw4w9WgXcQ'
};

async function main() {
    try {
        console.log("Fetching data...");
        const response = await axios.get(CONFIG.baseUrl, {
            params: {
                url: CONFIG.videoUrl,
                apikey: CONFIG.apiKey
            }
        });

        const data = response.data;
        if (data.status) {
            console.log(data);
        } else {
            console.log("Error:", data.message);
        }
    } catch (err) {
        console.error("Error:", err.message);
    }
}

main();

Node.js (ES Module)

modern-node

For modern projects (`"type": "module"` in package.json). Uses `import`.

import axios from 'axios';

const API_URL = 'https://xayz-ytdl.vercel.app/api/youtube';
const API_KEY = 'xayztech-YOUR_KEY_HERE';
const URL = 'https://www.youtube.com/watch?v=dQw4w9WgXcQ';

try {
    const { data } = await axios.get(API_URL, {
        params: { url: URL, apikey: API_KEY }

    if (data.status) {
        console.log(data);
    } else {
        console.error("Failed:", data.message);
    }
} catch (error) {
    console.error("Network Error:", error.message);
}

Node.js (Native Fetch)

no-dependencies

No `npm install` needed. Works on Node v18+.

const ENDPOINT = 'https://xayz-ytdl.vercel.app/api/youtube';
const KEY = 'xayztech-YOUR_KEY_HERE';
const VIDEO = 'https://www.youtube.com/watch?v=dQw4w9WgXcQ';

async function download() {
    const url = `${ENDPOINT}?url=${encodeURIComponent(VIDEO)}&apikey=${KEY}`;
    const res = await fetch(url);
    const json = await res.json();
    console.log(json);
}

download();

Python (Requests)

synchronous

The standard way. Requires `pip install requests`.

import requests

API_URL = "https://xayz-ytdl.vercel.app/api/youtube"
API_KEY = "xayztech-YOUR_KEY_HERE"
VIDEO = "https://www.youtube.com/watch?v=dQw4w9WgXcQ"

def main():
    params = { "url": VIDEO, "apikey": API_KEY }
    try:
        r = requests.get(API_URL, params=params)
        print(r.json())
    except Exception as e:
        print(e)

if __name__ == "__main__":
    main()

Python (Aiohttp)

asynchronous

For high-performance async apps. `pip install aiohttp`.

import aiohttp
import asyncio

URL = "https://xayz-ytdl.vercel.app/api/youtube"
KEY = "xayztech-YOUR_KEY_HERE"
VID = "https://www.youtube.com/watch?v=dQw4w9WgXcQ"

async def fetch_video():
    params = {"url": VID, "apikey": KEY}
    async with aiohttp.ClientSession() as session:
        async with session.get(URL, params=params) as resp:
            data = await resp.json()
            print(data)

asyncio.run(fetch_video())

Go (Golang)

backend 
package main
import (
    "fmt"
    "io/ioutil"
    "net/http"
    "net/url"
)

func main() {
    base := "https://xayz-ytdl.vercel.app/api/youtube"
    key := "xayztech-YOUR_KEY_HERE"
    video := "https://www.youtube.com/watch?v=dQw4w9WgXcQ"

    u, _ := url.Parse(base)
    q := u.Query()
    q.Set("url", video)
    q.Set("apikey", key)
    u.RawQuery = q.Encode()

    resp, _ := http.Get(u.String())
    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
