API Key Manager

Node.js (CommonJS)

server-side

Standard for older Node.js projects. Uses `require` and `axios`.

// 1. Install axios: npm install axios
const axios = require('axios');

const CONFIG = {
    baseUrl: 'https://xayz-ytdl.vercel.app/api/youtube',
    apiKey: 'xayztech-YOUR_KEY_HERE',
    videoUrl: 'https://www.youtube.com/watch?v=dQw4w9WgXcQ'
};

async function main() {
    try {
        console.log("Fetching data...");
        
        // Making the Request
        const response = await axios.get(CONFIG.baseUrl, {
            params: {
                url: CONFIG.videoUrl,
                apikey: CONFIG.apiKey
            }
        });

        const data = response.data;

        if (data.status) {
            console.log(`\nTitle: ${data.data.title}`);
            console.log(`Server: ${data.server_used}`);
            
            console.log("\nDownloads:");
            data.data.downloads.forEach(item => {
                console.log(`[${item.format}] ${item.quality} -> ${item.url}`);
            });
        } else {
            console.log("API Error:", data.message);
        }

    } catch (err) {
        console.error("Error:", err.message);
    }
}

main();

Node.js (Native Fetch)

no-dependencies

No `npm install` needed. Works on Node v18+.

const ENDPOINT = 'https://xayz-ytdl.vercel.app/api/youtube';
const KEY = 'xayztech-YOUR_KEY_HERE';
const VIDEO = 'https://www.youtube.com/watch?v=dQw4w9WgXcQ';

async function download() {
    // Construct URL with Query Params manually
    const url = `${ENDPOINT}?url=${encodeURIComponent(VIDEO)}&apikey=${KEY}`;

    const res = await fetch(url);
    const json = await res.json();

    if (json.status) {
        console.log(JSON.stringify(json.data, null, 2));
    } else {
        console.log("Error:", json.message);
    }
}

download();

Python (Requests)

synchronous

The standard way. Requires `pip install requests`.

import requests

API_URL = "https://xayz-ytdl.vercel.app/api/youtube"
API_KEY = "xayztech-YOUR_KEY_HERE"
VIDEO = "https://www.youtube.com/watch?v=dQw4w9WgXcQ"

def main():
    params = {
        "url": VIDEO,
        "apikey": API_KEY
    }

    try:
        print("Sending request...")
        r = requests.get(API_URL, params=params, timeout=30)
        
        data = r.json()

        if data.get("status"):
            info = data["data"]
            print(f"Title: {info['title']}")
            print(f"Duration: {info['duration']}")
            
            for dl in info['downloads']:
                print(f"[{dl['format'].upper()}] {dl['url']}")
        else:
            print(f"API Error: {data.get('message')}")

    except Exception as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()

Python (Aiohttp)

asynchronous

For high-performance async apps (Discord bots, FastAPI). `pip install aiohttp`.

import aiohttp
import asyncio

URL = "https://xayz-ytdl.vercel.app/api/youtube"
KEY = "xayztech-YOUR_KEY_HERE"
VID = "https://www.youtube.com/watch?v=dQw4w9WgXcQ"

async def fetch_video():
    params = {"url": VID, "apikey": KEY}
    
    async with aiohttp.ClientSession() as session:
        async with session.get(URL, params=params) as resp:
            data = await resp.json()
            
            if data['status']:
                print(f"Found: {data['data']['title']}")
            else:
                print("Failed")

# Run loop
asyncio.run(fetch_video())

PHP (cURL)

robust

Recommended for production PHP.

<?php
$baseUrl = "https://xayz-ytdl.vercel.app/api/youtube";
$key = "xayztech-YOUR_KEY_HERE";
$vid = "https://www.youtube.com/watch?v=dQw4w9WgXcQ";

$query = http_build_query(['url' => $vid, 'apikey' => $key]);
$finalUrl = $baseUrl . '?' . $query;

$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, $finalUrl);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_TIMEOUT, 30);
curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);

$res = curl_exec($ch);
curl_close($ch);

$json = json_decode($res, true);

if ($json['status']) {
    echo "Title: " . $json['data']['title'];
    print_r($json['data']['downloads']);
} else {
    echo "Error: " . $json['message'];
}
?>

PHP (file_get_contents)

simple

Easiest method, but ensure `allow_url_fopen` is enabled in php.ini.

<?php
$baseUrl = "https://xayz-ytdl.vercel.app/api/youtube";
$key = "xayztech-YOUR_KEY_HERE";
$vid = "https://www.youtube.com/watch?v=dQw4w9WgXcQ";

// Manual query string construction
$url = "$baseUrl?url=" . urlencode($vid) . "&apikey=$key";

$response = file_get_contents($url);

if($response) {
    $data = json_decode($response, true);
    if($data['status']) {
        echo "Found: " . $data['data']['title'];
    }
}
?>

Go (Golang)

backend 
package main

import (
    "fmt"
    "io/ioutil"
    "net/http"
    "net/url"
)

func main() {
    base := "https://xayz-ytdl.vercel.app/api/youtube"
    key := "xayztech-YOUR_KEY_HERE"
    video := "https://www.youtube.com/watch?v=dQw4w9WgXcQ"

    u, _ := url.Parse(base)
    q := u.Query()
    q.Set("url", video)
    q.Set("apikey", key)
    u.RawQuery = q.Encode()

    resp, err := http.Get(u.String())
    if err != nil { panic(err) }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
