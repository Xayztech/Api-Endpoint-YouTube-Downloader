Documentation

Welcome to the Xayz Tech YTDL API Documentation. This API allows developers to programmatically fetch direct download links for YouTube videos including MP4 (Video) and MP3 (Audio) formats.

Features

  • Hybrid Engine: Automatically switches between Invidious instances and BotCahx API if one fails.
  • Auto Conversion: Supports direct MP3 conversion.
  • No Ads: Clean JSON response without ad redirects.
  • CORS Enabled: Can be called from client-side Javascript.

API Endpoints

GET

Query Parameters

Param Required Description
url Yes The full YouTube video URL.
apikey Yes Your access key. Can also be sent via header x-api-key.

Error Handling

Standard HTTP status codes are used to indicate success or failure.

  • 200 OK : Request successful.
  • 400 Bad Request : Missing URL or Invalid URL format.
  • 401 Unauthorized : Invalid or missing API Key.
  • 503 Service Unavailable : All backend servers (Invidious & BotCahx) are busy or down.

Node.js Integration

const axios = require('axios');

const API_URL = 'https://xayz-ytdl.vercel.app/api/youtube';
const API_KEY = 'sk-pro-example';
const VIDEO_URL = 'https://www.youtube.com/watch?v=dQw4w9WgXcQ';

async function download() {
    try {
        const response = await axios.get(API_URL, {
            params: {
                url: VIDEO_URL,
                apikey: API_KEY
            }
        });

        const data = response.data;
        if(data.status) {
            console.log(`Title: ${data.data.title}`);
            
            data.data.downloads.forEach(item => {
                console.log(`[${item.type}] ${item.url}`);
            });
        }
    } catch (error) {
        console.error("Error:", error.response ? error.response.data : error.message);
    }
}

download();

Python Integration

Requires requests library.

import requests

BASE_URL = "https://xayz-ytdl.vercel.app/api/youtube"
API_KEY = "sk-pro-example"
VIDEO = "https://www.youtube.com/watch?v=dQw4w9WgXcQ"

try:
    params = {"url": VIDEO, "apikey": API_KEY}
    response = requests.get(BASE_URL, params=params)
    data = response.json()

    if data.get("status"):
        print(f"Title: {data['data']['title']}")
        print(f"Source: {data['server_used']}")
        
        for item in data['data']['downloads']:
            print(f"Link ({item['format']}): {item['url']}")
    else:
        print("Error:", data.get("message"))

except Exception as e:
    print("Exception:", e)

PHP Integration

Using cURL or file_get_contents.

<?php

$baseUrl = "https://xayz-ytdl.vercel.app/api/youtube";
$apiKey = "sk-pro-example";
$videoUrl = "https://www.youtube.com/watch?v=dQw4w9WgXcQ";

$query = http_build_query([
    'url' => $videoUrl,
    'apikey' => $apiKey
]);

// Initialize cURL
$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, $baseUrl . '?' . $query);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
$response = curl_exec($ch);
curl_close($ch);

$data = json_decode($response, true);

if ($data['status']) {
    echo "Video Found: " . $data['data']['title'] . "\n";
    foreach ($data['data']['downloads'] as $dl) {
        echo "Download " . $dl['format'] . ": " . $dl['url'] . "\n";
    }
} else {
    echo "Failed: " . $data['message'];
}
?>

Go (Golang)

package main

import (
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
    "net/url"
)

func main() {
    base := "https://xayz-ytdl.vercel.app/api/youtube"
    apiKey := "sk-pro-example"
    video := "https://www.youtube.com/watch?v=dQw4w9WgXcQ"

    u, _ := url.Parse(base)
    q := u.Query()
    q.Set("url", video)
    q.Set("apikey", apiKey)
    u.RawQuery = q.Encode()

    resp, err := http.Get(u.String())
    if err != nil { panic(err) }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    
    // You can parse JSON to struct here, printing raw for simplicity
    fmt.Println(string(body))
}

Java

Using generic HttpURLConnection.

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URL;

public class YTDownloader {
    public static void main(String[] args) {
        try {
            String apiKey = "sk-pro-example";
            String video = "https://www.youtube.com/watch?v=dQw4w9WgXcQ";
            String baseUrl = "https://xayz-ytdl.vercel.app/api/youtube?apikey=" + apiKey + "&url=" + video;

            URL url = new URL(baseUrl);
            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
            conn.setRequestMethod("GET");

            BufferedReader in = new BufferedReader(new InputStreamReader(conn.getInputStream()));
            String inputLine;
            StringBuffer content = new StringBuffer();
            while ((inputLine = in.readLine()) != null) {
                content.append(inputLine);
            }
            in.close();
            
            System.out.println(content.toString());
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
